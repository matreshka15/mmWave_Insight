    def send_config(self, config_file):
        """å‘é€é…ç½®æ–‡ä»¶"""

    def read_frame(self):
        """è¯»å–ä¸€å¸§æ•°æ®"""
        # æŸ¥æ‰¾é­”æ•°
        magic = b'\x02\x01\x04\x03\x06\x05\x08\x07'

            if byte == magic[0:1]:

                    break

    def parse_tlv(self, data):
        """è§£æ TLV æ ¼å¼æ•°æ®"""
        offset = 0
        detections = []

        while offset < len(data):
            tlv_type, tlv_length = struct.unpack('<II', data[offset:offset+8])
            offset += 8
